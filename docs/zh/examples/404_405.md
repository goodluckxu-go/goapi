## [<<](examples.md) 如何自定义404和405状态码返回
返回都是根据程序模块定义的，不同子程序模块分开定义
### 404返回
~~~go
func main() {
	api := goapi.GoAPI(true)
	api.NoRoute(func(ctx *goapi.Context) {
		ctx.Writer.WriteHeader(http.StatusNotFound)
		_, _ = ctx.Writer.Write([]byte("main not find"))
	})
	admin:=api.Child("/admin", "/admin")
	{
		admin.NoRoute(func(ctx *goapi.Context) {
			ctx.Writer.WriteHeader(http.StatusNotFound)
			_, _ = ctx.Writer.Write([]byte("admin not find"))
		})
		admin.AddMiddleware(nil)
	}
	user:=api.Child("/user", "/v1")
	{
		user.NoRoute(func(ctx *goapi.Context) {
			ctx.Writer.WriteHeader(http.StatusNotFound)
			_, _ = ctx.Writer.Write([]byte("user not find"))
		})
		user.AddMiddleware(nil)
	}
}
~~~
### 405返回
~~~go
func main() {
	api := goapi.GoAPI(true)
	api.HandleMethodNotAllowed = true // 开启method返回，不开启设置NoMethod无效
	api.NoMethod(func(ctx *goapi.Context) {
		ctx.Writer.WriteHeader(http.StatusNotFound)
		_, _ = ctx.Writer.Write([]byte("main not method"))
	})
	admin:=api.Child("/admin", "/admin")
	{
		admin.HandleMethodNotAllowed = true // 开启method返回，不开启设置NoMethod无效
		admin.NoMethod(func(ctx *goapi.Context) {
			ctx.Writer.WriteHeader(http.StatusNotFound)
			_, _ = ctx.Writer.Write([]byte("admin not method"))
		})
		admin.AddMiddleware(nil)
	}
	user:=api.Child("/user", "/v1")
	{
		user.HandleMethodNotAllowed = true // 开启method返回，不开启设置NoMethod无效
		user.NoMethod(func(ctx *goapi.Context) {
			ctx.Writer.WriteHeader(http.StatusNotFound)
			_, _ = ctx.Writer.Write([]byte("user not method"))
		})
		user.AddMiddleware(nil)
	}
}
~~~
